# TanStack Table Integration

TanStack Table (formerly React Table) provides powerful table functionality with sorting, filtering, pagination, and more. This guide shows how to integrate TanStack Table with Tabula Lens for advanced table features and performance optimization.

## Installation

First, install TanStack Table:

```bash
npm install @tanstack/react-table
# or
pnpm add @tanstack/react-table
# or
yarn add @tanstack/react-table
```

## Basic Setup

### Table Definition

Define your table structure with TanStack Table:

```tsx
import { useReactTable, getCoreRowModel, getSortedRowModel } from '@tanstack/react-table';

interface User {
  id: number;
  name: string;
  email: string;
  created_at: string;
}

function EnhancedDataTable({ data }: { data: { columns: string[]; rows: User[] } }) {
  const columns = [
    {
      accessorKey: 'id',
      header: 'ID',
      cell: (info: any) => info.getValue(),
    },
    {
      accessorKey: 'name',
      header: 'Name',
      cell: (info: any) => info.getValue(),
    },
    {
      accessorKey: 'email',
      header: 'Email',
      cell: (info: any) => info.getValue(),
    },
    {
      accessorKey: 'created_at',
      header: 'Created At',
      cell: (info: any) => new Date(info.getValue()).toLocaleDateString(),
    },
  ];

  const table = useReactTable({
    data: data.rows,
    columns,
    getCoreRowModel: getCoreRowModel(),
    getSortedRowModel: getSortedRowModel(),
  });

  return (
    <table>
      <thead>
        {table.getHeaderGroups().map((headerGroup) => (
          <tr key={headerGroup.id}>
            {headerGroup.headers.map((header) => (
              <th key={header.id}>
                {header.isPlaceholder
                  ? null
                  : (
                    <div
                      {...{
                        onClick: header.column.getToggleSortingHandler(),
                        style: { cursor: 'pointer' },
                      }}
                    >
                      {flexRender(
                        header.column.columnDef.header,
                        header.getContext()
                      )}
                      {header.column.getIsSorted() === 'asc' ? ' ↑' : null}
                      {header.column.getIsSorted() === 'desc' ? ' ↓' : null}
                    </div>
                  )}
              </th>
            ))}
          </tr>
        ))}
      </thead>
      <tbody>
        {table.getRowModel().rows.map((row) => (
          <tr key={row.id}>
            {row.getVisibleCells().map((cell) => (
              <td key={cell.id}>
                {flexRender(cell.column.columnDef.cell, cell.getContext())}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
}
```

### Custom Sorting

Implement custom sorting logic:

```tsx
const columns = [
  {
    accessorKey: 'status',
    header: 'Status',
    sortingFn: (rowA: any, rowB: any) => {
      const order = { active: 0, pending: 1, inactive: 2 };
      return order[rowA.original.status] - order[rowB.original.status];
    },
  },
];
```

## Pagination

### Client-Side Pagination

Implement client-side pagination:

```tsx
import { getPaginationRowModel } from '@tanstack/react-table';

const table = useReactTable({
  data,
  columns,
  getCoreRowModel: getCoreRowModel(),
  getPaginationRowModel: getPaginationRowModel(),
  initialState: {
    pagination: {
      pageSize: 20,
    },
  },
});

// Usage
<div>
  <button
    onClick={() => table.previousPage()}
    disabled={!table.getCanPreviousPage()}
  >
    Previous
  </button>
  <span>
    Page {table.getState().pagination.pageIndex + 1} of{' '}
    {table.getPageCount()}
  </span>
  <button
    onClick={() => table.nextPage()}
    disabled={!table.getCanNextPage()}
  >
    Next
  </button>
</div>
```

### Server-Side Pagination

Implement server-side pagination with Tabula Lens:

```tsx
import { useReactTable, getCoreRowModel } from '@tanstack/react-table';

function ServerSidePaginatedTable() {
  const [pagination, setPagination] = useState({ pageIndex: 0, pageSize: 20 });
  const [data, setData] = useState({ rows: [], total: 0 });

  useEffect(() => {
    fetchData(pagination.pageIndex + 1, pagination.pageSize).then(setData);
  }, [pagination]);

  const table = useReactTable({
    data: data.rows,
    columns,
    pageCount: Math.ceil(data.total / pagination.pageSize),
    getCoreRowModel: getCoreRowModel(),
    state: {
      pagination,
    },
    onPaginationChange: setPagination,
    manualPagination: true,
  });

  return <TableRenderer table={table} />;
}
```

## Performance Optimization

### Virtualization

Use virtualization for large datasets:

```tsx
import { useVirtualizer } from '@tanstack/react-virtual';

function VirtualizedTable({ data }: { data: { rows: any[] } }) {
  const tableContainerRef = useRef<HTMLDivElement>(null);

  const table = useReactTable({
    data: data.rows,
    columns,
    getCoreRowModel: getCoreRowModel(),
  });

  const { rows } = table.getRowModel();

  const rowVirtualizer = useVirtualizer({
    count: rows.length,
    getScrollElement: () => tableContainerRef.current,
    estimateSize: () => 50,
    overscan: 5,
  });

  return (
    <div ref={tableContainerRef} style={{ height: '500px', overflow: 'auto' }}>
      <div style={{ height: `${rowVirtualizer.getTotalSize()}px` }}>
        {rowVirtualizer.getVirtualItems().map((virtualRow) => {
          const row = rows[virtualRow.index];
          return (
            <div
              key={row.id}
              style={{
                position: 'absolute',
                top: 0,
                left: 0,
                width: '100%',
                height: `${virtualRow.size}px`,
                transform: `translateY(${virtualRow.start}px)`,
              }}
            >
              {row.getVisibleCells().map((cell) => (
                <span key={cell.id}>
                  {flexRender(cell.column.columnDef.cell, cell.getContext())}
                </span>
              ))}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```

### Lazy Loading

Implement lazy loading for large datasets:

```tsx
function LazyLoadedTable() {
  const [data, setData] = useState<any[]>([]);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);

  const loadMore = async () => {
    if (loading || !hasMore) return;
    setLoading(true);
    const newData = await fetchData(data.length, 50);
    setData([...data, ...newData]);
    setHasMore(newData.length === 50);
    setLoading(false);
  };

  return (
    <div>
      <Table data={data} />
      {hasMore && (
        <button onClick={loadMore} disabled={loading}>
          {loading ? 'Loading...' : 'Load More'}
        </button>
      )}
    </div>
  );
}
```

## Integration with DatabaseViewer

### Custom DataTable Component

Replace the default DataTable with TanStack Table:

```tsx
import { DatabaseViewer } from '@tabula-lens/react';
import { useReactTable, getCoreRowModel } from '@tanstack/react-table';

function TanStackDataTable({ data, sort, onSortChange }: any) {
  const columns = data.columns.map((col: string) => ({
    accessorKey: col,
    header: col.charAt(0).toUpperCase() + col.slice(1),
  }));

  const table = useReactTable({
    data: data.rows,
    columns,
    getCoreRowModel: getCoreRowModel(),
    state: {
      sorting: sort ? [sort] : [],
    },
    onSortingChange: (updater) => {
      const newSort = updater instanceof Function ? updater([]) : updater;
      onSortChange(newSort[0]);
    },
    getSortedRowModel: getSortedRowModel(),
  });

  return (
    <table>
      <thead>
        {table.getHeaderGroups().map((headerGroup) => (
          <tr key={headerGroup.id}>
            {headerGroup.headers.map((header) => (
              <th key={header.id}>
                {header.isPlaceholder
                  ? null
                  : (
                    <div
                      onClick={header.column.getToggleSortingHandler()}
                      style={{ cursor: 'pointer' }}
                    >
                      {flexRender(
                        header.column.columnDef.header,
                        header.getContext()
                      )}
                      {header.column.getIsSorted() === 'asc' ? ' ↑' : null}
                      {header.column.getIsSorted() === 'desc' ? ' ↓' : null}
                    </div>
                  )}
              </th>
            ))}
          </tr>
        ))}
      </thead>
      <tbody>
        {table.getRowModel().rows.map((row) => (
          <tr key={row.id}>
            {row.getVisibleCells().map((cell) => (
              <td key={cell.id}>
                {flexRender(cell.column.columnDef.cell, cell.getContext())}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
}

function App() {
  return (
    <DatabaseViewer
      path="/api/tabula-lens"
      dataComponent={TanStackDataTable}
    />
  );
}
```

### Hybrid Approach

Combine DatabaseViewer's features with TanStack Table:

```tsx
import { DatabaseViewer, useDatabaseData } from '@tabula-lens/react';
import { useReactTable, getCoreRowModel } from '@tanstack/react-table';

function HybridDatabaseViewer() {
  const { data, loading, error } = useDatabaseData({
    path: '/api/tabula-lens',
    table: 'users',
  });

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;

  const table = useReactTable({
    data: data?.rows || [],
    columns: data?.columns.map((col) => ({
      accessorKey: col,
      header: col,
    })) || [],
    getCoreRowModel: getCoreRowModel(),
  });

  return (
    <div>
      <DatabaseViewer path="/api/tabula-lens" tableSelector="dropdown" />
      <TanStackTable table={table} />
    </div>
  );
}
```

## Advanced Features

### Row Selection

Add row selection capabilities:

```tsx
import { getCoreRowModel, useReactTable } from '@tanstack/react-table';

function SelectableTable({ data }: { data: { rows: any[] } }) {
  const [rowSelection, setRowSelection] = useState({});

  const table = useReactTable({
    data: data.rows,
    columns,
    getCoreRowModel: getCoreRowModel(),
    state: {
      rowSelection,
    },
    onRowSelectionChange: setRowSelection,
    enableRowSelection: true,
  });

  const selectedRows = table.getFilteredSelectedRowModel().rows;

  return (
    <div>
      <TableRenderer table={table} />
      <div>Selected {selectedRows.length} rows</div>
    </div>
  );
}
```

## Best Practices

1. **Column Definition**: Use memoized column definitions to prevent re-renders
2. **Performance**: Use virtualization for large datasets
3. **Type Safety**: Leverage TypeScript for type-safe column definitions
4. **Accessibility**: Ensure table headers and interactions are accessible
5. **Responsive Design**: Make tables responsive for mobile devices
6. **Error Handling**: Handle empty states and loading states gracefully
7. **Testing**: Test table components and interactions thoroughly
8. **Server-Side**: Use server-side pagination for large datasets

## Troubleshooting

### Performance Issues

If table performance is slow:

```tsx
// Enable row virtualization
import { useVirtualizer } from '@tanstack/react-virtual';

// Memoize columns
const columns = useMemo(() => [...], []);

// Reduce re-renders
const table = useReactTable({
  // ... options
  enableRowVirtualization: true,
});
```

### Sorting Not Working

If sorting isn't working:

```tsx
// Ensure sorting model is enabled
const table = useReactTable({
  // ... options
  getSortedRowModel: getSortedRowModel(),
  enableSorting: true,
});
```

### Column Resizing Issues

If column resizing isn't working:

```tsx
// Ensure column resizing is enabled

  // ... options

  columnResizeMode: 'onChange',
});
```
